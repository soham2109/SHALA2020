def SGD(X,y,lr = 0.01,epochs=100,batch_size = 1):
  ################################################
  ####write your code here######################## 
  #################################################       
    w,b = np.random.randn(),0.1
    mse = []
    c = [i for i in range(0,len(X),batch_size)]
    for _ in range(epochs):
        idx = random.choice(c)
        x,y_ = X[idx:idx+batch_size],y[idx:idx+batch_size]
        grad_w = 0
        grad_b = 0
        e = 0
        for i,j in zip(x,y_):
            y_pred = w*i+b
            # calculate gradient
            grad_w += 2 * (y_pred - j)*i
            grad_b += 2 * (y_pred-j)
            e += (y_pred-j)**2
        # update rule
        w -= lr*grad_w/batch_size
        b -= lr*grad_b/batch_size
        mse.append(e/batch_size)
    return w, b, mse

Train a SVM classifier using gradient descent and return a Weight Matrix which is a numpy array of length (N + 1) where N is dimension of training samples. You can refer to Fig. 1 in this paper for implementation. You can add arguments to svm_function according to your implementation.

In [13]:
def svm_function(x, y, epoch, l_rate):
    ###################################
    ### Write your code here###########
    ###################################
    x = np.insert(x,0,1,axis=1)
    w = np.zeros(x.shape[1])
    for i in range(epoch):
        ind= np.random.choice(x.shape[0])
        x_i,y_i = x[ind],y[ind]
        eta_i = 1/(l_rate*(i+1))
        score = np.dot(w,x_i)
        if y_i*score < 1:
            w = (1-eta_i*l_rate)*w + eta_i*y_i*x_i
        else:
            w = (1-eta_i*l_rate)*w
    return w
